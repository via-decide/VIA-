camera.position.z = 5;

// Particles
const COUNT = window.innerWidth < 600 ? 600 : 1200;
const positions = new Float32Array(COUNT * 3);
const colors = new Float32Array(COUNT * 3);
const gold = new THREE.Color('#c8a84b');
const goldLight = new THREE.Color('#e8c96a');
const dim = new THREE.Color('#2a2620');

for (let i = 0; i < COUNT; i++) {
  positions[i * 3]     = (Math.random() - 0.5) * 20;
  positions[i * 3 + 1] = (Math.random() - 0.5) * 30;
  positions[i * 3 + 2] = (Math.random() - 0.5) * 10 - 3;
  const c = Math.random() < 0.15 ? goldLight : Math.random() < 0.4 ? gold : dim;
  colors[i * 3]     = c.r;
  colors[i * 3 + 1] = c.g;
  colors[i * 3 + 2] = c.b;
}

const geo = new THREE.BufferGeometry();
geo.setAttribute('position', new THREE.BufferAttribute(positions, 3));
geo.setAttribute('color', new THREE.BufferAttribute(colors, 3));

const mat = new THREE.PointsMaterial({
  size: 0.04,
  vertexColors: true,
  transparent: true,
  opacity: 0.7,
  sizeAttenuation: true,
});

const particles = new THREE.Points(geo, mat);

// Grid lines
const lineMat = new THREE.LineBasicMaterial({ color: 0x1a1508, transparent: true, opacity: 0.4 });
for (let i = -10; i <= 10; i += 2) {
  const hGeo = new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(-12, i, -5), new THREE.Vector3(12, i, -5)]);
  const vGeo = new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(i * 1.2, -15, -5), new THREE.Vector3(i * 1.2, 15, -5)]);
  scene.add(new THREE.Line(hGeo, lineMat));
  scene.add(new THREE.Line(vGeo, lineMat));
}

let scrollY = 0;
let targetScrollY = 0;
let mouseX = 0, mouseY = 0;

window.addEventListener('scroll', () => { targetScrollY = window.scrollY; });
window.addEventListener('mousemove', e => {

  mouseY = (e.clientY / window.innerHeight - 0.5) * 2;

  if (e.touches[0]) {

  particles.rotation.y = frame * 0.0003 + mouseX * 0.05;

function addTilt(selector) {
  document.querySelectorAll(selector).forEach(card => {
    card.addEventListener('mousemove', e => {
      const rect = card.getBoundingClientRect();
      const cx = rect.left + rect.width / 2;
      const cy = rect.top + rect.height / 2;
      const dx = (e.clientX - cx) / rect.width;
      const dy = (e.clientY - cy) / rect.height;
      card.style.transform = `rotateY(${dx * 8}deg) rotateX(${-dy * 5}deg) translateZ(8px)`;
      card.style.transition = 'transform 0.1s ease';
    });
    card.addEventListener('mouseleave', () => {
      card.style.transform = '';
      card.style.transition = 'transform 0.4s ease';
    });
  });
}
